function leFmtDate(?string $date): string
{
    if (!$date) {
        return '—';
    }

    $ts = strtotime($date);

    return ($ts === false) ? '—' : date('m-d-Y', $ts);
}

function leFmtDateLong(?string $date): string
{
    if (!$date) {
        return '—';
    }

    $ts = strtotime($date);

    return ($ts === false) ? '—' : date('F j, Y', $ts);
}

function lePadNum($value): string
{
    $value = trim((string) $value);

    if ($value === '') {
        return '';
    }

    if (ctype_digit($value) && strlen($value) < 2) {
        return str_pad($value, 2, '0', STR_PAD_LEFT);
    }

    return $value;
}

function leBuildNumberRange(int $max): array
{
    $items = [];
    $i = 1;

    while ($i <= $max) {
        $key = ($i < 10) ? '0' . $i : (string) $i;
        $items[$key] = 0;
        $i++;
    }

    return $items;
}

function leBuildLabelList(int $max): array
{
    return array_keys(leBuildNumberRange($max));
}

function leCommaList(array $items): string
{
    $clean = [];
    $i = 0;

    while ($i < count($items)) {
        $item = trim((string) $items[$i]);
        if ($item !== '') {
            $clean[] = $item;
        }
        $i++;
    }

    return empty($clean) ? '—' : implode(', ', $clean);
}

function leTopKeysByValue(array $counts, int $limit, bool $ascending = false): array
{
    $copy = $counts;

    if ($ascending) {
        asort($copy, SORT_NUMERIC);
    } else {
        arsort($copy, SORT_NUMERIC);
    }

    return array_slice(array_keys($copy), 0, $limit);
}

function leHasMeaningfulJackpot($value): bool
{
    $value = trim((string) $value);

    if ($value === '' || strtolower($value) === 'n/a') {
        return false;
    }

    return is_numeric($value) && (float) $value > 0;
}

function leSafeJson($value): string
{
    return json_encode($value, JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE);
}

function leNumberToInt(string $value): int
{
    $value = trim($value);
    return ($value !== '' && ctype_digit($value)) ? (int) $value : 0;
}

function leConsecutiveCount(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $v) {
        $n = is_int($v) ? $v : leNumberToInt((string) $v);
        if ($n > 0) {
            $sorted[] = $n;
        }
    }
    sort($sorted, SORT_NUMERIC);
    $count = 0;
    for ($i = 1; $i < count($sorted); $i++) {
        if ($sorted[$i] - $sorted[$i - 1] === 1) {
            $count++;
        }
    }
    return $count;
}

function leArithmeticPatternScore(array $numbers): int
{
    $sorted = [];
    foreach ($numbers as $v) {
        $n = is_int($v) ? $v : leNumberToInt((string) $v);
        if ($n > 0) {
            $sorted[] = $n;
        }
    }
    sort($sorted, SORT_NUMERIC);
    if (count($sorted) < 4) {
        return 0;
    }
    $same = 0;
    for ($i = 2; $i < count($sorted); $i++) {
        if (($sorted[$i - 1] - $sorted[$i - 2]) === ($sorted[$i] - $sorted[$i - 1])) {
            $same++;
        }
    }
    return $same;
}

function leNormalizePlayForGame(array $numbers, int $min, int $max, int $limit): array
{
    $ints = [];
    $seen = [];
    foreach ($numbers as $v) {
        $n = is_int($v) ? $v : leNumberToInt((string) $v);
        if ($n < $min || $n > $max || isset($seen[$n])) {
            continue;
        }
        $seen[$n] = true;
        $ints[] = $n;
        if (count($ints) >= $limit) {
            break;
        }
    }
    sort($ints, SORT_NUMERIC);
    $out = [];
    foreach ($ints as $n) {
        $out[] = ($n < 10) ? '0' . $n : (string) $n;
    }
    return $out;
}

function leBuildPlayFromPoolsForGame(array $pools, array $fallback, int $min, int $max, int $limit): array
{
    $picked = [];
    $seen   = [];
    foreach ($pools as $pool) {
        foreach ($pool as $v) {
            $n = is_int($v) ? $v : leNumberToInt((string) $v);
            if ($n < $min || $n > $max || isset($seen[$n])) {
                continue;
            }
            $seen[$n] = true;
            $picked[]  = $n;
            if (count($picked) >= $limit) {
                break 2;
            }
        }
    }
    if (count($picked) < $limit) {
        foreach ($fallback as $v) {
            $n = is_int($v) ? $v : leNumberToInt((string) $v);
            if ($n < $min || $n > $max || isset($seen[$n])) {
                continue;
            }
            $seen[$n] = true;
            $picked[]  = $n;
            if (count($picked) >= $limit) {
                break;
            }
        }
    }
    return leNormalizePlayForGame($picked, $min, $max, $limit);
}

function leBuildCombinationProfileForGame(array $play, array $latestMain, int $numberMax, int $mainCount): array
{
    $ints = [];
    foreach ($play as $v) {
        $n = is_int($v) ? $v : leNumberToInt((string) $v);
        if ($n > 0) {
            $ints[] = $n;
        }
    }
    sort($ints, SORT_NUMERIC);
    $low = (int) round($numberMax * 0.5);
    $odd = 0;
    $lo  = 0;
    foreach ($ints as $n) {
        if ($n % 2 !== 0) {
            $odd++;
        }
        if ($n <= $low) {
            $lo++;
        }
    }
    $sum         = array_sum($ints);
    $spread      = !empty($ints) ? (max($ints) - min($ints)) : 0;
    $consecutive = leConsecutiveCount($ints);
    $latestLook  = [];
    foreach ($latestMain as $v) {
        $k = trim((string) $v);
        if (strlen($k) === 1 && ctype_digit($k)) {
            $k = '0' . $k;
        }
        $latestLook[$k] = true;
    }
    $repeated = 0;
    foreach ($ints as $n) {
        $k = ($n < 10) ? '0' . $n : (string) $n;
        if (isset($latestLook[$k])) {
            $repeated++;
        }
    }
    $t1 = (int) round($numberMax * 0.34);
    $t2 = (int) round($numberMax * 0.54);
    $density = 'Wide spread';
    if ($spread <= $t1) {
        $density = 'High cluster density';
    } elseif ($spread <= $t2) {
        $density = 'Moderate cluster density';
    }
    return [
        'odd_even'        => $odd . '/' . (count($ints) - $odd),
        'low_high'        => $lo . '/' . (count($ints) - $lo),
        'sum'             => $sum,
        'spread'          => $spread,
        'consecutive'     => $consecutive,
        'repeated'        => $repeated,
        'cluster_density' => $density,
    ];
}

function leCrowdRiskProfileForGame(array $play, array $favoriteNumbers, int $numberMax): array
{
    $ints = [];
    foreach ($play as $v) {
        $n = is_int($v) ? $v : leNumberToInt((string) $v);
        if ($n > 0) {
            $ints[] = $n;
        }
    }
    sort($ints, SORT_NUMERIC);
    $favLook  = [];
    foreach ($favoriteNumbers as $f) {
        $favLook[(int) $f] = true;
    }
    $under31  = 0;
    $favHits  = 0;
    foreach ($ints as $n) {
        if ($n <= 31) {
            $under31++;
        }
        if (isset($favLook[$n])) {
            $favHits++;
        }
    }
    $spread      = !empty($ints) ? (max($ints) - min($ints)) : 0;
    $consecutive = leConsecutiveCount($ints);
    $visual      = leArithmeticPatternScore($ints) >= 2;
    $score       = 0;
    $reasons     = [];
    if ($under31 >= 4) {
        $score += 2;
        $reasons[] = 'birthday-heavy composition';
    } elseif ($under31 >= 3) {
        $score += 1;
        $reasons[] = 'leaning toward under-31 numbers';
    }
    if ($consecutive >= 2) {
        $score += 2;
        $reasons[] = 'sequence concentration';
    } elseif ($consecutive === 1) {
        $score += 1;
        $reasons[] = 'minor sequence pattern';
    }
    if ($visual) {
        $score += 1;
        $reasons[] = 'obvious spacing pattern';
    }
    $t1 = (int) round($numberMax * 0.39);
    $t2 = (int) round($numberMax * 0.54);
    if ($spread <= $t1) {
        $score += 2;
        $reasons[] = 'tight clustering';
    } elseif ($spread <= $t2) {
        $score += 1;
        $reasons[] = 'moderate clustering';
    }
    if ($favHits >= 2) {
        $score += 1;
        $reasons[] = 'cultural favorite overlap';
    }
    $label = 'Low crowd risk';
    if ($score >= 5) {
        $label = 'High crowd risk';
    } elseif ($score >= 3) {
        $label = 'Medium crowd risk';
    }
    $explanation = !empty($reasons)
        ? 'Signals: ' . implode(', ', $reasons) . '.'
        : 'Signals: broad spread and lower overlap with common crowd patterns.';
    return ['score' => $score, 'label' => $label, 'explanation' => $explanation];
}

  function hasDataset(values) {
    var i;

  function commonBarOptions(horizontal) {
    return {
      responsive: true,
      maintainAspectRatio: false,
      animation: false,
      indexAxis: horizontal ? 'y' : 'x',
      plugins: {
        legend: { display: false },
        tooltip: { enabled: true }
      },
      scales: horizontal ? {
        x: {
          beginAtZero: true,
          ticks: { precision: 0 },
          grid: { color: 'rgba(10,26,51,.08)' }

          grid: { color: 'rgba(10,26,51,.08)' }
        }

  function ensureFallbackMessage(canvas, message) {
    var frame;
    var note;
    if (!canvas) {
      return;
    }
    frame = canvas.parentNode;
    if (!frame) {
      return;
    }
    note = document.createElement('div');
    note.className = 'skai-empty';
    note.style.padding = '12px';
    note.textContent = message;
    frame.innerHTML = '';
    frame.appendChild(note);
  }

  function createChart(id, type, labels, values, color, options) {
    var canvas;
    var context;
    var width;
    if (createdCharts[id]) {
      return true;
    }
    canvas = document.getElementById(id);
    if (!canvas) {
      return true;
    }
    if (!window.Chart) {
      return false;
    }
    width = canvas.parentNode && canvas.parentNode.offsetWidth ? canvas.parentNode.offsetWidth : canvas.offsetWidth;
    if (!width || width < 40) {
      return false;
    }
    if (!hasDataset(values)) {
      ensureFallbackMessage(canvas, 'No chart data available for this view.');
      createdCharts[id] = true;
      return true;
    }
    context = canvas.getContext('2d');
    createdCharts[id] = new window.Chart(context, {
      type: type,
      data: {
        labels: labels,
        datasets: [{
          data: values,
          borderWidth: 0,
          borderRadius: 8,
          maxBarThickness: 18,
          backgroundColor: color
        }]
      },
      options: options
    });
    return true;
  }

        allReady = false;
      }
    }

    if (!allReady && chartRetryCount < maxChartRetries) {
      chartRetryCount++;
      window.setTimeout(renderCharts, 250);
    }
  }

  function renderTracked(selector, wrapId, chipClass, emptyText) {

      empty.textContent = emptyText;
      wrap.appendChild(empty);
      return;
    }

    for (i = 0; i < items.length; i++) {
      chip = document.createElement('span');
      chip.className = 'skai-chip ' + chipClass;
      chip.textContent = items[i];
      wrap.appendChild(chip);
    }
  }

  function bindStrategySaving() {
    var status = document.getElementById('strategySaveStatus');
    var buttons = document.querySelectorAll('.js-save-strategy');
    var i;

    function sanitizeText(value) {
      return String(value || '').replace(/[^0-9,\sA-Za-z]/g, '').replace(/\s+/g, ' ').trim();
    }

    var strategyStore = {
      key: 'le_strategies',
      save: function (payload) {
        var safeName = sanitizeText(payload.name);
        var safeMain = sanitizeText(payload.main);
        var safeBonus = sanitizeText(payload.bonus);
        var existing = [];
        var stored;
        var entry;
        var i;
        try {
          stored = window.localStorage.getItem(this.key);
          if (stored) {
            existing = JSON.parse(stored) || [];
            if (!Array.isArray(existing)) {
              existing = [];
            }
          }
        } catch (e) {
          return { ok: false, message: 'Browser storage is not available on this device.' };
        }
        for (i = 0; i < existing.length; i++) {
          if (existing[i] && existing[i].name === safeName) {
            return { ok: true, message: '"' + safeName + '" is already in your saved strategies.' };
          }
        }
        entry = { name: safeName, main: safeMain, bonus: safeBonus };
        existing.push(entry);
        try {
          window.localStorage.setItem(this.key, JSON.stringify(existing));
          return { ok: true, message: '"' + safeName + '" saved to your browser. Your strategies are available until browser data is cleared.' };
        } catch (e) {
          return { ok: false, message: 'Unable to save. Browser storage may be full or disabled.' };
        }
      }
    };

    function setStatus(message) {
      if (status) {
        status.textContent = message;
      }
    }

    for (i = 0; i < buttons.length; i++) {
      buttons[i].addEventListener('click', function () {
        var payload = {
          name: sanitizeText(this.getAttribute('data-strategy') || 'Strategy'),
          main: sanitizeText(this.getAttribute('data-main') || ''),
          bonus: sanitizeText(this.getAttribute('data-bonus') || '')
        };
        var result = strategyStore.save(payload);
        setStatus((result.ok ? '' : 'Notice: ') + result.message);
      });
    }
  }
